```java
/**
 * Manages native QLever lifecycle and error conversion.
 * Arena-scoped memory management with QLeverStatus → QLeverException mapping.
 */
public final class NativeHandle<T extends QLeverEngine> implements AutoCloseable {

    private final Arena arena = Arena.ofShared();
    private final MemorySegment engineHandle;
    private final QLeverStatus lastStatus;

    // Factory methods
    public static NativeHandle<QLeverEngine> create() throws QLeverException {
        MemorySegment handle = jextract.qlever_ffi.qlever_engine_create(arena);
        QLeverStatus status = extractStatus(handle);
        if (status.code() != 0) {
            throw new QLeverException("Engine creation failed: " + status.message(), status);
        }
        return new NativeHandle<>(handle, status);
    }

    // Error conversion layer
    private QLeverException convertError(QLeverStatus status) {
        return switch (status.code()) {
            case 1 -> new QLeverParseException(status.message(), status);
            case 2 -> new QLeverSemanticException(status.message(), status);
            case 3 -> new QLeverRuntimeException(status.message(), status);
            default -> new QLeverException(status.message(), status);
        };
    }

    // Memory segment management
    public MemorySegment getEngineHandle() {
        return engineHandle;
    }

    public Arena getArena() {
        return arena;
    }

    @Override
    public void close() {
        jextract.qlever_ffi.qlever_engine_destroy(engineHandle);
        arena.close();
    }
}
```

```cpp
        int code;        // 0=success, 1=parse, 2=semantic, 3=runtime

    };

            return {2, strdup(e.what())};

            return {3, strdup(e.what())};
        }
    }
```

### Layer 2: ErlangNode (Typed Bridge)

```java
/**
 * Manages Erlang distribution protocol with ei.h bindings.
 * Unix domain socket, ETF encoding, error term conversion.
 */
public final class ErlangNode implements AutoCloseable {

    // Core connection management
    public void connect(String targetNode, String cookie) throws ErlangConnectionException;
    public void connect(String targetHost, int targetPort, String cookie) throws ErlangConnectionException;

    // RPC with automatic error term conversion
    public ErlTerm rpc(String module, String function, List<ErlTerm> args) throws ErlangException {
        byte[] argBytes = ErlTermCodec.encodeArgs(args);
        MemorySegment resultBuf = callArena.allocate(ei_h.EI_X_BUFF_LAYOUT);

        int rc = ei_h.ei_rpc(cnode, fd, modSeg, funSeg, argSeg, args.length, resultBuf);
        if (rc != 0) {
            throw new ErlangRpcException(module, function, "ei_rpc failed: " + rc);
        }

        byte[] resultBytes = extractBytesFromXbuff(resultBuf);
        ei_h.ei_x_free(resultBuf);

        // Convert Erlang error terms to Java exceptions
        ErlTerm result = ErlTermCodec.decodeRpcResult(resultBytes);
        return convertErlangError(result);
    }

    // Error conversion layer
    private ErlTerm convertErlangError(ErlTerm term) throws ErlangException {
        if (term instanceof ErlTuple t && t.elements().length >= 2) {
            ErlTerm first = t.elements()[0];
            ErlTerm second = t.elements()[1];

            if (first instanceof ErlAtom a && "badrpc".equals(a.value())) {
                if (second instanceof ErlTuple inner && inner.elements().length >= 2) {
                    ErlAtom exitAtom = (ErlAtom) inner.elements()[0];
                    if ("EXIT".equals(exitAtom.value())) {
                        ErlTerm reason = inner.elements()[1];
                        throw new ErlangExitException(reason);
                    }
                }
                throw new ErlangRpcException("Remote call failed: " + second);
            }
        }
        return term;
    }

    // Send/receive with typed encoding
    public void send(String registeredName, ErlTerm message) throws ErlangSendException;
    public ErlMessage receive() throws ErlangReceiveException;
}
```

### Layer 3: gen_server Interface (Erlang)

```erlang
%% process_mining_bridge gen_server
%% Handles capability execution and registry management
-behaviour(gen_server).

-export([start_link/0, import_ocel_json_path/2, slim_link_ocel/2]).
-export([init/1, handle_call/3, handle_cast/3, handle_info/2, terminate/2, code_change/3]).

-record(state, {
    registry :: ets:tid(),  % Mnesia-backed capability registry
    node :: pid(),          % Connected ErlangNode Java process
    timeout :: timeout()    % Operation timeout
}).

%% Public API
start_link() ->
    gen_server:start_link({local, ?MODULE}, ?MODULE, [], []).

import_ocel_json_path(#{path := Path}, Timeout) ->
    gen_server:call(?MODULE, {import_ocel_json_path, Path}, Timeout).

slim_link_ocel(#{ocel_id := OcelId}, Timeout) ->
    gen_server:call(?MODULE, {slim_link_ocel, OcelId}, Timeout).

%% gen_server callbacks
init([]) ->
    % Initialize Mnesia registry that survives restarts
    Registry = ets:new(capability_registry, [
        public, named_table, {heir, node(), []},
        {read_concurrency, true}, {write_concurrency, true}
    ]),

    % Connect to JVM ErlangNode
    {ok, NodePid} = erlang_node:start(),

    State = #state{registry = Registry, node = NodePid, timeout = 30000},
    {ok, State}.

handle_call({import_ocel_json_path, Path}, _From, State) ->
    % Call Rust NIF via Java boundary
    case rust4pm:import_ocel_json_path(Path) of
        {ok, OcelId} ->
            % Store in Mnesia registry
            ets:insert(State#state.registry, {OcelId, now()}),
            {reply, {ok, OcelId}, State};
        {error, Reason} ->
            {reply, {error, Reason}, State}
    end;

handle_call({slim_link_ocel, OcelId}, _From, State) ->
    case ets:lookup(State#state.registry, OcelId) of
        [{OcelId, _}] ->
            % Call Rust NIF
            case rust4pm:slim_link_ocel(OcelId) of
                {ok, SlimId} ->
                    ets:insert(State#state.registry, {SlimId, OcelId, now()}),
                    {reply, {ok, SlimId}, State};
                {error, Reason} ->
                    {reply, {error, Reason}, State}
            end;
        [] ->
            {reply, {error, not_found}, State}
    end.
```

### Layer 2: NIF Boundary (Rust → Erlang)

```rust
// rust4pm/src/lib.rs - NIF boundary
use std::ffi::{CStr, CString};
use std::os::raw::c_char;
use std::ptr;

// #[no_mangle] pub extern "C" fn rust4pm_parse_ocel2_json(
//     json_ptr: *const c_char,
//     json_len: usize,
// ) -> ParseResult;

#[no_mangle]
pub extern "C" fn rust4pm_import_ocel_json_path(path_ptr: *const c_char) -> ParseResult {
    let path = unsafe { CStr::from_ptr(path_ptr).to_string_lossy() };

    match std::fs::read_to_string(&path) {
        Ok(json) => {
            let log = OcelLog::from_json(&json)?;
            let uuid = Uuid::new_v4();

            // Store in Rust heap
            RUST4PM_STATE.with(|state| {
                state.borrow_mut().logs.insert(uuid, log);
            });

            ParseResult::new(uuid)
        }
        Err(e) => ParseResult::error(e.to_string()),
    }
}

#[no_mangle]
pub extern "C" fn rust4pm_slim_link(uuid: Uuid) -> SlimLinkResult {
    RUST4PM_STATE.with(|state| {
        if let Some(log) = state.borrow().logs.get(&uuid) {
            let slim = SlimOcel::from_ocel(log);
            let slim_uuid = Uuid::new_v4();

            state.borrow_mut().slim_logs.insert(slim_uuid, slim);

            SlimLinkResult::new(slim_uuid)
        } else {
            SlimLinkResult::error("OcelId not found")
        }
    })
}
```

### ErlTerm Sealed Interface Hierarchy (Java)

```java
/**
 * Sealed interface for all Erlang External Term Format types.
 * Exhaustive pattern matching supported across domains.
 */
public sealed interface ErlTerm
    permits ErlAtom, ErlLong, ErlBinary, ErlList, ErlTuple {

    // Common interface for all Erlang terms
    String encode();
    static ErlTerm decode(byte[] bytes) throws ErlangDecodeException;
}

// Specific implementations
public final record ErlAtom(String value) implements ErlTerm {
    @Override public String encode() { return "\"" + value + "\""; }
}

public final record ErlLong(long value) implements ErlTerm {
    @Override public String encode() { return Long.toString(value); }
}

public final record ErlBinary(byte[] data) implements ErlTerm {
    @Override public String encode() {
        return "<<\"" + Base64.getEncoder().encodeToString(data) + "\">>";
    }
}

public final record ErlList(List<ErlTerm> elements, ErlTerm tail) implements ErlTerm {
    @Override public String encode() {
        String items = elements.stream().map(ErlTerm::encode).collect(Collectors.joining(", "));
        return "[" + items + (tail == null ? "" | "|" + tail.encode()) + "]";
    }
}

public final record ErlTuple(List<ErlTerm> elements) implements ErlTerm {
    @Override public String encode() {
        String items = elements.stream().map(ErlTerm::encode).collect(Collectors.joining(", "));
        return "{" + items + "}";
    }
}
```

### CallPattern Routing Logic

```java
/**
 * Routes NativeCall triples to appropriate bridge layers
 * based on callPattern registry requirements.
 */
public final class BridgeRouter {

    public ErlTerm routeCall(NativeCall call) throws BridgeException {
        return switch (call.callPattern()) {
            case "jvm" -> routeToJvm(call);
            case "beam" -> routeToBeam(call);
            case "direct" -> routeDirect(call); // Never used, architectural escape valve
            default -> throw new BridgeException("Unknown call pattern: " + call.callPattern());
        };
    }

    private ErlTerm routeToJvm(NativeCall call) {
        // Sub-10ns in-process Panama FFM to libqlever_ffi.so
        try (NativeHandle<QLeverEngine> handle = NativeHandle.create()) {
            return switch (call.function()) {
                case "qlever_ask" -> {
                    QLeverEngine engine = new QLeverEngine(handle);
                    ErlAtom result = ErlAtom.valueOf(engine.ask(call.arguments().get("sparql")));
                    yield result;
                }
                // ... other JVM-pattern calls
            };
        }
    }

    private ErlTerm routeToBeam(NativeCall call) throws BridgeException {
        // Crosses Boundary A: JVM → Unix socket → BEAM → NIF Boundary B
        try (ErlangNode node = new ErlangNode("yawl@localhost")) {
            node.connect("erlang@localhost", 5671, "secret");

            // Convert Java arguments to ErlTerm
            List<ErlTerm> erlArgs = call.arguments().values().stream()
                .map(this::convertToErlangTerm)
                .toList();

            return node.rpc(call.module(), call.function(), erlArgs);
        }
    }
}
```

```java
// Java error hierarchy for Boundary A
public sealed interface ErlangException extends Exception
    permits ErlangConnectionException, ErlangRpcException, ErlangExitException {

    ErlTerm getErlangTerm();
}

public final class ErlangConnectionException extends ErlangException {
    private final String target;
    private final int errorCode;

    public ErlangConnectionException(String target, String message, int errorCode) {
        super(message);
        this.target = target;
        this.errorCode = errorCode;
    }

    @Override public ErlTerm getErlangTerm() {
        return ErlTuple.of(
            ErlAtom.of("badrpc"),
            ErlTuple.of(
                ErlAtom.of("noproc"),
                ErlAtom.of(target)
            )
        );
    }
}

public final class ErlangExitException extends ErlangException {
    private final ErlTerm reason;

    public ErlangExitException(ErlTerm reason) {
        super("Erlang exit: " + reason.encode());
        this.reason = reason;
    }

    @Override public ErlTerm getErlangTerm() {
        return ErlTuple.of(
            ErlAtom.of("badrpc"),
            ErlTuple.of(ErlAtom.of("EXIT"), reason)
        );
    }
}
```

```rust
// NIF boundary fault containment

pub extern "C" fn rust4pm_token_replay(
    ocel_uuid: Uuid,
    pn_uuid: Uuid,

            unsafe { *result = r; }

            unsafe { *result = error_term; }

        }
    }
```

### Memory Flow Across Boundaries

```java
// JVM → BEAM: ErlTerm encoding/decoding flow
public class ErlangNode {
    public ErlTerm rpc(String module, String function, List<ErlTerm> args) {
        // 1. Java objects → ErlTerm hierarchy
        List<ErlTerm> encodedArgs = args.stream()
            .map(ErlTerm::encode)
            .toList();

        // 2. ErlTerm → ETF bytes (UTF-8)
        byte[] etfBytes = ErlTermCodec.encode(encodedArgs);

        // 3. ei_rpc → Unix domain socket → Erlang BEAM
        MemorySegment resultBuf = callArena.allocate(ei_h.EI_X_BUFF_LAYOUT);
        int rc = ei_h.ei_rpc(cnode, fd, modSeg, funSeg, argSeg, args.length, resultBuf);

        // 4. ETF bytes → ErlTerm hierarchy (reverse flow)
        byte[] resultBytes = extractBytesFromXbuff(resultBuf);
        return ErlTermCodec.decodeRpcResult(resultBytes);
    }
}

// BEAM → NIF: Handle passing flow
gen_server:handle_call({import_ocel_json_path, Path}, _From, State) ->
    % 1. File path → Erlang binary
    JsonData = case file:read_file(Path) of
        {ok, Data} -> Data;
        {error, _} -> exit({file_error, Path})
    end,

    % 2. Erlang binary → NIF call (UTF-8 bytes)
    case rust4pm_import_ocel_json_path(JsonData) of
        {ok, Uuid} ->
            % 3. UUID binary → Mnesia registry → durable handle
            ets:insert(State#state.registry, {Uuid, now()}),
            {reply, {ok, Uuid}, State}
    end.
```

## Implementation Guidelines

### Java 25 Requirements

- **Records**: Use for all immutable data (OcelId, SlimOcelId, etc.)
- **Sealed interfaces**: For ErlTerm hierarchy and error types
- **Virtual threads**: `Thread.ofVirtual()` for per-case processing
- **Structured concurrency**: `StructuredTaskScope` for parallel operations
- **MemorySegment**: Arena-based memory management for native calls

### Erlang Requirements

- **gen_server**: One-for-one supervision for fault isolation
- **Mnesia**: Tables with `disc_copies` for restart persistence
- **ei.h**: Direct Unix domain socket connections (bypass EPMD)
- **Hot reload**: Code server integration for seamless updates

### Rust Requirements

- **NIF boundary**: `#[no_mangle] pub extern "C"` functions
- **UUID handles**: Opaque object identifiers in Rust heap
- **Error handling**: Result<T, Error> with proper conversion to Erlang terms
- **Memory safety**: No unsafe code except at NIF boundary

### Quality Gates

| Aspect | Target | Measurement |
|--------|---------|-------------|
| **Latency** | <1ms (JVM-QLever), <50µs (JVM-BEAM) | Microbenchmarks |
| **Throughput** | 10K QPS per domain | Load testing |
| **Fault isolation** | 0 JVM crashes from Rust faults | Chaos engineering |
| **Memory** | 100MB heap per domain | Memory profiling |
| **Hot reload** | <100ms service restart | A/B testing |

---

This architecture enforces the invariant: **A rust4pm fault cannot reach the JVM.** The isolation guarantee is achieved through OS process separation, not defensive programming.
