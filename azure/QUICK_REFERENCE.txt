1. Install Azure CLI
   macOS:    brew install azure-cli
   Linux:    curl -sL https://aka.ms/InstallAzureCLIDeb | sudo bash
   Windows:  choco install azure-cli

2. Authenticate
   $ az login
   $ az account show

3. Deploy
   $ cd /home/user/yawl/azure
   $ chmod +x deploy.sh
   $ ./deploy.sh production eastus

4. Verify
   $ az deployment group show \
     --resource-group yawl-production \
     --query properties.outputs

5. Access
   https://<app-service-name>.azurewebsites.net/resourceService/

  az account show                                   # Current account info
  az account list                                   # List all accounts

DEPLOYMENT
  ./deploy.sh production eastus                     # Bash script
  .\deploy.ps1 -Environment production              # PowerShell script
  az deployment group create --resource-group ...   # Manual deployment

VALIDATION
  az deployment group validate --template-file azuredeploy.json
  jq empty azuredeploy.json                         # Validate JSON

MONITORING
  az webapp log tail --resource-group <rg> --name <app>
  az monitor metrics list --resource <resource-id>
  az monitor app-insights metrics show --app <app>

DATABASE
  psql -h <db-fqdn> -U <user>@<db> -d <database>
  az postgres server show --resource-group <rg> --name <db>
  az postgres server firewall-rule list --server <db>

SCALING
  az appservice plan update --sku P3V2 --resource-group <rg>
  az webapp config set --resource-group <rg> --name <app>

CLEANUP
  az group delete --name yawl-production --yes --no-wait
